Gene expression data is highly dynamic and temporal. In this assignment, you will compare two deep learning approaches—an LSTM model and an Attention-based model—to classify gene expression patterns. You will use three synthetic gene expression datasets, each with a different sequence length (100, 200, and 300 time steps). Additionally, you will perform grid search hyperparameter tuning to find the best model configurations for each architecture.

Dataset

You will work with three datasets:
● gene_expression_data_100.csv: 100 time steps
● gene_expression_data_200.csv: 200 time steps
● gene_expression_data_300.csv: 300 time steps
Each dataset contains 1,000 samples with 10 gene features per time step. The outcome (binary) indicates a specific gene expression pattern that you need to classify.

def prepare_data(data):
  y = data['Outcome'].values.astype(np.float32)
  x = data.drop('Outcome', axis=1).values.astype(np.float32)
  num_samples = x.shape[0]
  num_genes = 10
  num_time_steps = x.shape[1] // num_genes
  x = x.reshape(num_samples, num_time_steps, num_genes)
  return train_test_split(x, y, test_size=0.2, random_state=42)

# loading the datasets with our fxn
x_train_100, x_test_100, y_train_100, y_test_100 = prepare_data(data_100)

x_train_300, x_test_300, y_train_300, y_test_300 = prepare_data(data_300)

def lstm_model(units=64, dropout_rate=0.3, learning_rate=0.001, sequence_length=None, feature_dim=None):
  model = Sequential([
      tf.keras.Input(shape=(x_train_100.shape[1], x_train_100.shape[2])),
      LSTM(units),
      Dropout(dropout_rate),
      Dense(1, activation='sigmoid')
      ])
  model.compile(optimizer=Adam(learning_rate=learning_rate),
                loss='binary_crossentropy',
                metrics=['accuracy'])
  return model

# doing grid search
lstm_hyperparameters = {
    'units': [32, 64],
    'dropout_rate': [0.2, 0.3],
    'learning_rate': [0.001, 0.0005],
    'batch_size': [32, 64]
}

best_val_acc = 0
best_lstm_model = None
best_lstm_parameters = {}

for units in lstm_hyperparameters['units']:
  for dropout in lstm_hyperparameters['dropout_rate']:
    for lr in lstm_hyperparameters['learning_rate']:
      for batch in lstm_hyperparameters['batch_size']:
        model = lstm_model(units=units, dropout_rate=dropout, learning_rate=lr)
        history = model.fit(x_train_100, y_train_100,
                            validation_data=(x_test_100, y_test_100),
                            epochs=10, batch_size=batch, verbose=0)
        val_acc = history.history['val_accuracy'][-1]
        if val_acc > best_val_acc:
          best_val_acc = val_acc
          best_lstm_model = model
          best_lstm_parameters = {'units':units, 'dropout_rate': dropout,
                                  'learning_rate': lr, 'batch_size': batch}

print("Best LSTM Parameters: ", best_lstm_parameters)
print("Best Validation Accuracy: ", best_val_acc)

def attention_model(num_heads=2, ff_dim=64, dropout_rate=0.3, learning_rate=0.001, sequence_length=100, feature_dim=10):
  inputs = tf.keras.Input(shape=(sequence_length, feature_dim))
  attn_output = MultiHeadAttention(num_heads=num_heads, key_dim=feature_dim)(inputs, inputs)
  attn_output = Dropout(dropout_rate)(attn_output)
  out1 = Add() ([inputs, attn_output])
  out1 = LayerNormalization()(out1)

  # ffn
  ffn = tf.keras.Sequential([Dense(ff_dim, activation='relu'),
                             Dense(x_train_100.shape[2])])

  ffn_output= ffn(out1)
  ffn_output = Dropout(dropout_rate)(ffn_output)
  out2 = Add()([out1, ffn_output])
  out2 = LayerNormalization()(out2)

  x = GlobalAveragePooling1D()(out2)
  outputs = Dense(1, activation='sigmoid')(x)

  model = tf.keras.Model(inputs=inputs, outputs=outputs)
  model.compile(optimizer=Adam(learning_rate=learning_rate),
                loss='binary_crossentropy',
                metrics=['accuracy'])
  return model

def plot_training(history, title):
  plt.figure()
  plt.plot(history.history['accuracy'], label='Train Acc')
  plt.plot(history.history['val_accuracy'], label='Val Acc')
  plt.title(title + ' - Accuracy')
  plt.legend()
  plt.show()

  plt.figure()
  plt.plot(history.history['loss'], label='Train Loss')
  plt.plot(history.history['val_loss'], label='Val Loss')
  plt.title(title + ' - Loss')
  plt.legend()
  plt.show()

def train_todata(x_train, y_train, x_test, y_test, model_builder, params, model_type):
    model_build_params = {k: v for k, v in params.items() if k != 'batch_size'}
    sequence_length = x_train.shape[1]
    feature_dim = x_train.shape[2]
    model = model_builder(
        **model_build_params,
        sequence_length=sequence_length,
        feature_dim=feature_dim
        )

    history = model.fit(
        x_train, y_train,
        validation_data=(x_test, y_test),
        epochs=20,
        batch_size=params['batch_size']
        )
    plot_training(history, f"{model_type} - {x_train.shape[1]} steps")
    return history.history['val_accuracy'][-1]
